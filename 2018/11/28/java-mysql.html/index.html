发表于 2018-11-28 | 分类于 mysql

java程序取出 mysql 数据库的信息

配置

在项目中导入 mysql-connector-java-8.0.13.jar 包，下载网址：https://dev.mysql.com/downloads/connector/j/，在Select Operating System:                                                                                             
下选择 Platform Independent，下载 zip 包后解压，里面有相应的 jar 包，导入这个包就可以了。

代码

 public class TestJDBC {

       public static void main(String[] args) throws Exception{
        String driver = "com.mysql.jdbc.Driver";

        //"jdbc:mysql": 通用的

         //"localhost:3306/ceshi": localhost:主机名或ip地址; 3306:端口; ceshi:数据库的名字

           /*"?serverTimezone=UTC": (解决下面的错误，不要忘记加"?")
          * -->Caused by: com.mysql.cj.core.exceptions.InvalidConnectionAttributeException:
          *    The server time zone value '�й���׼ʱ��' is unrecognized or represents more than one time zone.
           *    You must configure either the server or JDBC driver (via the serverTimezone configuration property)
           *    to use a more specifc time zone value if you want to utilize time zone support.
         */

        String url = "jdbc:mysql://localhost:3306/ceshi?serverTimezone=UTC";
        String password="zwy";
        String username="root";            

        //注册数据库驱动(反射：通过一个类名将这个数据反向加载到网页中)，driver值是固定的
        Class.forName(driver);

        //获得Connection数据库连接
        Connection conn = DriverManager.getConnection(url, username, password);

           //获得数据库连接执行平台
        Statement state = conn.createStatement();

        //sql语句, "ceshi_table": table名字
        String sql = "select * from ceshi_table";

        //通过执行平台执行查的操作并返回结果
        ResultSet set = state.executeQuery(sql);

        while(set.next()){
            System.out.println(set.getString("id"));
            System.out.println(set.getString("name"));
        }

        state.close();
        conn.close();
    }
}
